# type 和 interface 的区别

https://www.typescriptlang.org/docs/handbook/2/everyday-types.html#functions interface 可继承的，可以通过重复定义 interface 对已有的 interface 进行字段的补充，type 在定义后不能再进行定义。
interface 只能用于定于对象，而不能定义原始类型；type 可以。

// An interface cannot extend a primitive type like 'string'; an interface can only extend named types and classes
interface P1 extends string {}
type P2 = string;
type P3 = { name: string };

# unknow、ant 和 never

unknow 更加安全，对一个 unknow 类型的执行操作会报错。

function f1(a: any) {
  a.b(); // OK
}
function f2(a: unknown) {
  a.b();   // Object is of type 'unknown'.
}

never 类型一般是函数中抛出异常或终止执行时的返回值。同时如果一个联合类型没有剩余的类型时

function fn(x: string | number) {
  if (typeof x === "string") {
    // do something
  } else if (typeof x === "number") {
    // do something else
  } else {
    x; // has type 'never'!
  }
}

# 类型操作

typescript 拥有很强大的类型系统，可以通过其他 type 创建 type。https://www.typescriptlang.org/docs/handbook/2/types-from-types.html

# 泛型构建

// 泛型类
class Lala<T extends string> {
  private tag: T;

  constructor(tag: T) {
    this.tag = tag;
  }
}
const la: Lala<string> = new Lala("lala");

interface Lengthwise {
  length: number;
}
 
function loggingIdentity<Type extends Lengthwise>(arg: Type): Type {
  console.log(arg.length); 
  return arg;
}

在泛型约束中定义类型参数

function getProperty<Type, Key extends keyof Type>(type: Type, key: Key) {
  return type[key];
}

type Person = {
  name: string;
  age: number;
};
const p1: Person = {
  name: "hen",
  age: 24,
};
getProperty(p1, "name");
getProperty(p1, "age");

# keyof 操作符

keyof 操作符接受一个 object 类型，生成这个 object 的一个 string 或 number 类型的属性值的联合类型。

type Point = {
  [n: string]: number;
};
// 如果一个对象有索引器，keyof 会返回 string | number 的联合类型
// 在 javscript 中可以通过 obj[0] 的形式取值，最终 obj[0] 会转化为 obj["0"]。所以这里会返回 string | number 的联合类型
type PointKeys = keyof Point;

# typeof 操作符

通过 typeof 操作符可以引用一个变量的类型。

const str = "str";
const str2: typeof str = "str";

在获取一个函数的返回类型时，可以结合 ReturnType 和 typeof

function fn(){
    return {name: 'fn', age: 24}
}
// 报错，fn 是一个值，却被当成一个类型处理
type FnReturn = ReturnType<fn>;
// 通过 typeof 操作符，获取 fn 的类型
type FnReturn = ReturnType<typeof fn>;

typeof 操作符，一般用于变量名和他的属性上

const obj = {
  name: "14",
};

type NameType = typeof obj.name;
type ObjType = typeof obj;

条件类型

interface Animal {
  live(): void;
}
interface Dog extends Animal {
  woof(): void;
}
 
type Example1 = Dog extends Animal ? number : string;
// 在条件类型中通过 infer 推断出类型
type Flatten<Type> = Type extends Array<infer Item> ? Item : Type;
// 通过 infer 推断泛型函数的返回值类型
type GetReturnType<Type> = Type extends (...args: never[]) => infer Return
  ? Return
  : never;
type Fn1ReturnType = GetReturnType<() => string>; // string
type Fn1ReturnType = GetReturnType<() => number>; // number

映射类型,基于另一个有索引器的类型上

type FeatureFlags = {
  darkMode: () => void;
  newUserProfile: () => void;
};

type OptionsFlags<Type> = {
  [Property in keyof Type]: boolean;
};

type BooleanHoreses = OptionsFlags<FeatureFlags>; 

// 修饰符的映射, readonly 和 ? 操作符可以在映射的过程中可用
type CreateMutable<Type> = {
  -readonly [Property in keyof Type]: Type[Property];
};
 
type LockedAccount = {
  readonly id: string;
  readonly name: string;
};
 
type UnlockedAccount = CreateMutable<LockedAccount>;
// 移除可选项
type Concrete<Type> = {
  [Property in keyof Type]-?: Type[Property];
};
 
type MaybeUser = {
  id: string;
  name?: string;
  age?: number;
};
 
type User = Concrete<MaybeUser>;
